// 直接调用DeepSeek API async function callAPI() { try { const response = await window.mysticalAI.callDeepSeekAPI( '请为我分析今日运势', 'fortune' ); console.log(response); } catch (error) { console.error('API调用失败:', error); } }

⚠️ 错误处理

常见错误及解决方案

  • • API密钥未设置: 请先访问设置页面配置API密钥
  • • 网络连接失败: 检查网络连接和API端点
  • • 配额不足: 检查DeepSeek账户余额
  • • 请求超时: 增加timeout设置或检查网络状况
// 完整的错误处理示例 async function safeAPICall(prompt, type) { try { const result = await window.mysticalAI.callDeepSeekAPI(prompt, type); return { success: true, data: result }; } catch (error) { console.error('API调用失败:', error); // 根据错误类型提供不同的处理 if (error.message.includes('API密钥')) { return { success: false, error: '请先设置API密钥', action: 'redirectToSettings' }; } else if (error.message.includes('网络')) { return { success: false, error: '网络连接失败，请稍后重试', action: 'retry' }; } else { return { success: false, error: '未知错误，请稍后重试', action: 'retry' }; } } }
